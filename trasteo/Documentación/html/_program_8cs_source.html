374 public static string GenerarWikiDataURL(string url, string animal)
375 {
376 // Buscar si el código de la entidad empieza por 'Q' y extraerlo
377 var match = Regex.Match(url, @"Q\d+");
378 if (match.Success)
379 {
380 return $"https://www.wikidata.org/wiki/{match.Value}";
381 }
382 else
383 {
384 Console.WriteLine($"No se pudo corregir la URL de Wikidata para {animal}");
385 return $"https://www.wikidata.org/wiki/Q";
386 }
